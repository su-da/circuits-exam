\begin{question}
    求图中电流源的功率（说明是实际吸收还是发出）。(10 分)
    \begin{figure}[H]
    \hfill
    \input fig01
    \end{figure}
    \begin{answer}
        电压源发出功率 $P_\mathrm{V} = 1 \times 1 = \unit{1}{\watt}$ (3
        分)，电流源电压(上正下负)为 $U = -2 \times 1 + 1 =
        \unit{-1}{\volt}$(4 分)，所以电流源吸收功率 $-1 \times 1 =
        \unit{-1}{\watt}$(3 分)。
    \end{answer}
\end{question}

\begin{question}
    求图中的电流 $I$。(12 分)
    \begin{figure}[H]
    \hfill
    \input fig02
    \end{figure}
    \begin{answer}
        回路电流法。取回路电流在三个网孔上，顺时针。左下角网孔电流为
        $I_1$，右下角网孔电流为 $I$，列回路方程。
        \begin{equation*}
        \left.\begin{aligned}
        (10+5+10)I-10I_1-5\times 1 &= -5 \\
        -10I+(10+10)I_1-10 \times 1 &= 10
        \end{aligned}\right\}
        \end{equation*}
        解得 $I=\unit{0.5}{\ampere}$，$I_1=\unit{1.25}{\ampere}$。
    \end{answer}
\end{question}

\begin{question}
    列写图中电路的结点电压方程并求解。(12 分)
    \begin{figure}[H]
    \hfill
    \input fig03
    \end{figure}
    \examvspace*{2cm}
    \begin{answer}
        取下面结点为参考零点，左上和右上分别为 $U_\mathrm{n1}$ 和
        $U_\mathrm{n2}$。列方程得。
        \begin{equation*}
        \left.\begin{aligned}
            (5+1+1+1)U_\mathrm{n1}-(1+1)U_\mathrm{n2} &= 5 - 1 \\
             -(1+1)U_\mathrm{n1}+(1+1+1)U_\mathrm{n2} &= 1 + 3
        \end{aligned}\right\}
        \end{equation*}
        解得 $U_\mathrm{n1}=\unit{1}{\volt}$，
        $U_\mathrm{n2}=\unit{2}{\volt}$。
    \end{answer}
\end{question}

\begin{question}
    求下图电路的戴维宁或诺顿等效电路。(12 分)
    \begin{figure}[H]
    \hfill
    \input fig04
    \end{figure}
    \examvspace*{7cm}
    \begin{answer}
        开路时有 $(2+4)I_1=4+2I_1$，解得 $I_1=\unit{1}{\ampere}$，
        $U_\mathrm{OC}=4I_1-2I_1=\unit{2}{\volt}$。短路时有方程
        \begin{equation*}
        \left.\begin{aligned}
            (2+4)I_1-4I_\mathrm{SC} &= 4+2I_1 \\
             -4I_1+(4+2)I_\mathrm{SC} &= -2I_1
        \end{aligned}\right\}
        \end{equation*}
        解得 $I_\mathrm{SC}=\unit{0.5}{\ampere}$，
        $R_\mathrm{eq}=\unit{4}{\ohm}$。
    \end{answer}
\end{question}

\begin{question}
    $R_\mathrm{L}$ 为多大时 $R_\mathrm{L}$ 可获得最大功率，
    求此最大功率。(12 分)
    \begin{figure}[H]
    \hfill
    \input fig05
    \end{figure}
    \examvspace*{6cm}
    \begin{answer}
        负载电阻去除之后部分的戴维宁等效电路
        $R_\mathrm{eq}=\unit{10}{\ohm}$，
        $U_\mathrm{OC}=\unit{10}{\volt}$，最大功率条件为
        $R_\mathrm{L}=R_\mathrm{eq}$，最大功率
        $P_\mathrm{max}=U_\mathrm{OC}^2/(4R_\mathrm{eq})=\unit{2.5}{\watt}$。
    \end{answer}
\end{question}

\begin{question}
    下图电路中开关打开以前电路已达稳态，$t = 0$ 时开关 $S$ 打开。求电容电流 $i_C(t)$。(12 分)
    \begin{figure}[H]
    \hfill
    \input fig06
    \end{figure}
    \examvspace*{8cm}
    \begin{answer}
        初始条件 $u_C(0_+)=u_C(0_-)=\unit{3}{\volt}$，最终稳态
        $u_C(\infty)=\unit{6}{\volt}$，$R_\mathrm{eq}=\unit{200}{\ohm}$
        ，时间常数 $\tau = R_\mathrm{eq}C=\unit{0.2}{\second}$。
        根据三要素法得 $u_C(t)=6-3\mathrm{e}^{-5t}\unit{}{\volt}$，微分得
        $i_C(t)=15\mathrm{e}^{-5t}\unit{}{\milli\ampere}$。
    \end{answer}
\end{question}

\begin{question}
    求下图电路的输入阻抗 $Z$ 和输入导纳 $Y$。(10 分)
    \begin{figure}[H]
    \hfill
    \input fig07
    \end{figure}
    \examvspace*{5cm}
    \begin{answer}
    容易求得 $Z=(2-\jimag 1)\unit{}{\ohm}$ ($6'$)，$Y=1/Z=(0.4+\jimag
    0.2)\unit{}{\siemens}$ ($4'$)。
    \end{answer}
\end{question}

\begin{question}
    列写图示电路的回路电流方程和结点电压方程。(20 分)
    \begin{figure}[H]
    \hfill
    \input fig08
    \end{figure}
    \examvspace*{5cm}
    \begin{answer}
    结点电压方程为
    \begin{equation*}\left.\begin{aligned}
    3\dot{U}_\mathrm{n1}-2\dot{I}-\dot{U}_\mathrm{n3} &= \dot{U}_\mathrm{S}
    \\
    -\dot{U}_\mathrm{n1}-\jimag
    0.1\times 2\dot{I}+\dot{U}_\mathrm{n3} &= 0 \\
    \dot{I} &= (\dot{U}_\mathrm{n3}-\dot{U}_\mathrm{n1})/1
    \end{aligned}\right\}\end{equation*}
    网孔方程为 ($\dot{I}_1$ 和 $\dot{I}_2$ 分别为下边两个网孔电流，顺时
    针，上边网孔电流逆时针)
    \begin{equation*}\left.\begin{aligned}
    2 \dot{I}_1 + \dot{I} &= \dot{U}_\mathrm{S} - 2 \dot{I} \\
    -\jimag 10 \dot{I} &= 2 \dot{I} \\
    \dot{I}_1 - \jimag 10 \dot{I}_2 + (2-\jimag 10)\dot{I} &= 0
    \end{aligned}\right\}\end{equation*}
    \end{answer}
\end{question}
