\begin{question} % 01
    求电路图中各电源的功率（说明是吸收还是发出）。(10 分)
    \begin{figure}[H]
    \hfill
    \input fig01
    \end{figure}
    \examvspace*{2cm}
    \begin{answer}
        电流源吸收功率 $P_\mathrm{I} = 1 \times 1 = \unit{1}{\watt}$ (3
        分)，电压源电流(向上)满足 $1/1\times 1 = I - 1$(4 分)，所以电压源发出功率 $I \times 1 = \unit{2}{\watt}$(3 分)。
    \end{answer}
\end{question}

\begin{question} % 02
    求下图所示电路的电流 $I$。(12 分)
    \begin{figure}[H]
    \hfill
    \input fig02
    \end{figure}
    \examvspace*{4cm}
    \begin{answer}
        用回路电流法。上边网孔顺时针，大小为 $\unit{1}{\ampere}$，
        左下角网孔顺时针大小为 $I$，横跨下面俩网孔电流顺时针，大小设为
        $I_1$。列方程如下：
        \begin{equation*}
        \left.\begin{aligned}
        (5+5+30)I+(5+5)I_1-5\times 1&=30\\
        (5+5)I+(5+5+20)I_1-(5+20)\times 1&=30-5
        \end{aligned}\right\}
        \end{equation*}
        解得 $I = \unit{0.5}{\ampere}$。
    \end{answer}
\end{question}

\begin{question} %03
    列写以下电路的结点电压方程并求解之。(12 分)
    \begin{figure}[H]
    \hfill
    \input fig03
    \end{figure}
    \examvspace*{7cm}
    \begin{answer}
        取下面结点为参考零点，左上和右上分别为 $U_\mathrm{n1}$ 和
        $U_\mathrm{n2}$，列方程如下。
        \begin{equation*}
        \left.\begin{aligned}
            (1/2+1/2)U_\mathrm{n1}-1/2U_\mathrm{n2}&=2I \\
           -1/2U_\mathrm{n1}+(1/2+1/4)U_\mathrm{n2}&=1 \\
                                                 I &= U_\mathrm{n2}/4
        \end{aligned}\right\}
        \end{equation*}
        解得 $U_\mathrm{n1}=U_\mathrm{n2}=\unit{4}{\volt}$，
        $I=\unit{1}{\ampere}$。
    \end{answer}
\end{question}

\begin{question} % 04
    求图示电路的戴维宁或诺顿等效电路。(12 分)
    \begin{figure}[H]
    \hfill
    \input fig04
    \end{figure}
    \examvspace*{7cm}
    \begin{answer}
        等效电阻为 $R_\mathrm{eq}=\unit{10}{\ohm}$，开路电压(上正下负)为

    \end{answer}
\end{question}

\begin{question} % 05
    下图电路的 $R$ 可变，问其为何值时可获得最大功率，最大功率为多少？(12 分)
    \begin{figure}[H]
    \hfill
    \input fig05
    \end{figure}
    \examvspace*{6cm}
    \begin{answer}
        去掉 $R$ 后的电路开路电压(上正下负)为 $U_\mathrm{OC}=4\times 4 +
        \frac{3}{6+3}24=\unit{24}{\volt}$，等效电路
        $R_\mathrm{eq}=6\parallel 3 + 4=\unit{6}{\ohm}$。最大功率条件
        $R=R_\mathrm{eq}=\unit{6}{\ohm}$，最大功率为
        $P_\mathrm{max}=U_\mathrm{OC}^2/(4R_\mathrm{eq})=\unit{24}{\watt}$。
    \end{answer}
\end{question}

\begin{question} %06
    图示电路原已稳定，$t = 0$ 时合上开关 $S$，求电感电流 $i_L(t)$。(12 分)
    \begin{figure}[H]
    \hfill
    \input fig06
    \end{figure}
    \examvspace*{7cm}
    \begin{answer}
        $i_L(\infty)=i_L(0_+)=i_L(0_-)=12/6=\unit{2}{\ampere}$，根

    \end{answer}
\end{question}

\begin{question} %07
    求下图电路的输入阻抗 $Z$ 和导纳 $Y$。(10 分)
    \begin{figure}[H]
    \hfill
    \input fig07
    \end{figure}
    \examvspace*{3cm}
    \begin{answer}
    $Z=(\jimag \omega L \dot{I}-r\dot{I})/\dot{I}=\jimag\omega L-r$
    ($6'$)，$Y=1/(\jimag\omega L-r)$ ($4'$)。
    \end{answer}
\end{question}

\begin{question} % 08
    给出图中电路的结点电压方程和回路电流方程。(20 分)
    \begin{figure}[H]
    \hfill
    \input fig08
    \end{figure}
    \examvspace*{7cm}
    \begin{answer}
    结点电压方程为
    \begin{equation*}
    \left.\begin{aligned}
        -\frac{1}{\jimag 5}10\Varangle{\unit{0}{\degree}}+\dot{U}_\mathrm{n2}-\dot{U}_\mathrm{n3}&=0\\
    -\dot{U}_\mathrm{n2}+\dot{U}_\mathrm{n3}&=-g\dot{U}_{\rm o}\\
    10\Varangle{\unit{0}{\degree}}-\dot{U}_\mathrm{n3}&=\dot{U}_{\rm o}
    \end{aligned}\right\}
    \end{equation*}
    网孔方程为
    \begin{equation*}
    \left.\begin{aligned}
    \jimag 5g\dot{U}_{\rm o}&=10\Varangle{\unit{0}{\degree}}\\
    \jimag 5\dot{I}_2+g\dot{U}_{\rm o}&=\dot{U}_{\rm o}
    \end{aligned}\right\}
    \end{equation*}
    \end{answer}
\end{question}
