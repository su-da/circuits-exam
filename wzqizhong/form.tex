\begin{question}
    求图示电路中各元件的功率，并说明实际是发出功率还是吸收功率。(10 分\defen)
    \begin{figure}[H]
    \hfill
    \input fig01
    \end{figure}
    \examvspace*{3cm}
    \begin{answer}
        电流源两端电压 $U = -1 \times 2 + 10 = \unit{8}{\volt}$，电阻吸收的功率 $P_1 = \unit{4}{\watt}$，电压源的功率 $P_2 = 10I = \unit{20}{\watt}$，发出功率 $\unit{20}{\watt}$；电流源功率 $P_3 = \unit{16}{\watt}$，吸收功率 $\unit{16}{\watt}$。
    \end{answer}
\end{question}

\begin{question}
    求电流 $I_\mathrm{AB}$。(12 分\defen)
    \begin{figure}[H]
    \hfill
    \input fig02
    \end{figure}
    \examvspace*{4cm}
    \begin{answer}
        电压源输出电流 \unit{4.5}{\ampere}，$I_\mathrm{AB}=\unit{1.5}{\ampere}$。
    \end{answer}
\end{question}

\begin{question}
求图所示电路的输入电阻 $R_\mathrm{in}$。(12 分\defen)
    %\examvspace{12pt}\\
    \begin{figure}[H]
    \hfill
    \input fig03
    \end{figure}
    \examvspace*{6cm}
    \begin{answer}
        在端口上加电压源 $u$，设端口电流为 $i$，可列写方程 $u = 2(i-u)+1i=3i-2u$，解得输入电阻为 $R_\mathrm{in} = \unit{1}{\ohm}$。
    \end{answer}
\end{question}

    \begin{answer}

    \end{answer}
\end{question}

\begin{question}
用叠加定理求图示电路中电阻支路的电流 $I_1$、$I_3$。(12 分\defen)
    \begin{figure}[H]
    \hfill
    \input fig05
    \end{figure}
    \examvspace*{6cm}
    \begin{answer}
        \unit{2}{\ampere} 电流源单独作用时为分别为 \unit{-1}{\ampere}、\unit{-0.5}{\ampere}。\unit{4}{\volt} 电压源单独作用时分别为 \unit{2}{\ampere}、\unit{1}{\ampere}。可解得 $I_1 = \unit{1}{\ampere}$，$I_3 = \unit{0.5}{\ampere}$。
    \end{answer}
\end{question}

\begin{question}
    $R_\mathrm{L}$ 可调，求 $R_\mathrm{L}$ 所能获得的最大功率。(12 分\defen)
    \begin{figure}[H]
    \hfill
    \input fig06
    \end{figure}
    \examvspace*{7cm}
    \begin{answer}
        解得开路电压 $U_\mathrm{OC} = \unit{-2}{\volt}$，$R_\mathrm{eq} = \unit{1}{\ohm}$，获得的最大功率为 \unit{1}{\watt}。
    \end{answer}
\end{question}
