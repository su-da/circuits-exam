\begin{question}
    求图示电路中的 $U_1$ 和 $U_2$。(5 分\defen)
    \begin{figure}[H]
    \hfill
    \input fig01
    \end{figure}
    % \examvspace*{0cm}
    \begin{answer}
        $U_1 = -2\times 5 = \unit{-10}{\volt}$，$U_2 = 10 + 15 = \unit{25}{\volt}$。
    \end{answer}
\end{question}

\begin{question}
    求图示电路中各元件的功率，并说明实际是发出功率还是吸收功率。(5 分\defen)
    \begin{figure}[H]
    \hfill
    \input fig02
    \end{figure}
    % \examvspace*{0cm}
    \begin{answer}
        电流源两端电压 $U = -1 \times 2 + 10 = \unit{8}{\volt}$，电阻吸收的功率 $P_1 = \unit{4}{\watt}$，电压源的功率 $P_2 = 10I = \unit{20}{\watt}$，发出功率 $\unit{20}{\watt}$；电流源功率 $P_3 = \unit{16}{\watt}$，吸收功率 $\unit{16}{\watt}$。
    \end{answer}
\end{question}

\begin{question}
    求图所示电路的输入电阻$ R_\mathrm{in}$。(10 分\defen)
    %\examvspace{12pt}\\
    \begin{figure}[H]
    \hfill
    \input fig04
    \end{figure}
    \examvspace*{3cm}
    \begin{answer}
        在端口上加电压源 $u$，设端口电流为 $i$，可列写方程
    \begin{equation*}
        \begin{aligned}
            u &= 2i + 8i_1 \\
            8i_1 &= 4i_1 + 4(i - i_1)
        \end{aligned}
    \end{equation*}
    解得输入电阻为 $R_\mathrm{in} = \unit{6}{\ohm}$。
    \end{answer}
\end{question}

\begin{question}
用回路电流法求图示电路中电压 $U_1$，$U_2$ 和电流 $I$。(10 分\defen)
    \begin{figure}[H]
    \hfill
    \input fig05
    \end{figure}
    \examvspace*{3cm}
    \begin{answer}
        $U_1 = 1\times 1 + 10 =\unit{11}{\volt}$，$U_2 = -1 \times 2 + 10 = \unit{8}{\volt}$，有 $-1 + 10/2 + I + 2 = 0$，得 $I = \unit{-6}{\ampere}$。
    \end{answer}
\end{question}

\begin{question}
    用两种方法(等效变换、回路法、结点法三选二)求电流 $I$ 和电压 $U$。(15 分\defen)
    \begin{figure}[H]
    \hfill
    \input fig09
    \end{figure}
    \examvspace*{6cm}
    \begin{answer}
        回路法：
    \begin{equation*}
        \begin{aligned}
            6I + 2\times 1 &= 8 - 3U \\
            U &= 2(1+I)+3U
        \end{aligned}
    \end{equation*}
    解得 $I = \unit{3}{\ampere}$，$U = \unit{-4}{\volt}$。结点法、等效变换略。
    \end{answer}
\end{question}

\begin{question}
用两种方法(等效变换、回路法、结点法三选二)求电流 $I$ 和电压 $U$。 (15 分\defen)
    \begin{figure}[H]
    \hfill
    \input fig10
    \end{figure}
    \examvspace*{6cm}
    \begin{answer}
        结点法：
    \begin{equation*}
        \begin{aligned}
            (1+1)U &= (5 - I)/1 -U \\
            -U &= 1I + I - 5
        \end{aligned}
    \end{equation*}
    解得 $I = \unit{2}{\ampere}$，$U = \unit{1}{\volt}$。回路法、等效变换略。
    \end{answer}
\end{question}
