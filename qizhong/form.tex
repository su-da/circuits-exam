\begin{question}
    当 $u_C = \mathrm{e}^{-2t} \unit{}{\volt}$ 时，求电压 $u$。(10 分)
    \begin{figure}[H]
        \hfill
        \input fig01
    \end{figure}
    \examvspace*{1cm}
    \begin{answer}
        $i = 6u_\mathrm{C}' = -12\mathrm{e}^{-2t}\unit{}{\ampere}$，
        $u = 2i + u_\mathrm{C} = -23\mathrm{e}^{-2t}\unit{}{\volt}$。
    \end{answer}
\end{question}

\begin{question}
    当 $i_L = \mathrm{e}^{-t} \unit{}{\ampere}$ 时，求 $i$。(10 分)
    %\examvspace{12pt}\\
    \begin{figure}[H]
    \hfill
    \input fig02
    \end{figure}
    \examvspace*{1cm}
    \begin{answer}
        $u = 6i_\mathrm{L}' = -6\mathrm{e}^{-t}\unit{}{\volt}$，
        $i = 3u + i_\mathrm{L} = -17\mathrm{e}^{-t}\unit{}{\ampere}$。
    \end{answer}
\end{question}

\begin{question}
    电路原已处于稳态，在 $t=0$ 时开关 $S$ 断开，求此时的 $i_L(t)$ 和
    $u(t)$。(10 分)

    \hfill
    \input fig07
    \end{figure}

    \begin{answer}
        $i_L(t)=2(1-\mathrm{e}^{-3t})\unit{}{\ampere}$。
    \end{answer}
\end{question}

\begin{question}
    图示电路中，开关合在 1 时已达稳态。$t=0$ 时开关由1合向2，求 $t>0$ 时
    的 $i_1(t)$。(15 分)
    \begin{figure}[H]
    \hfill
    \input fig08
    \end{figure}
    \examvspace*{18cm}
    \begin{answer}
        $i_\mathrm{L}(0^+) = i_\mathrm{L}(0^-) = \unit{-4}{\volt}$，
        $i_\mathrm{L}(\infty)=\unit{1.2}{\ampere}$，
        $i_1(\infty)=\unit{0.8}{\ampere}$，
        $i_1(0^+)=\unit{6}{\ampere}$，$\tau=\unit{0.01}{\second}$，
        所以 $i_1(t)=0.8+5.2\mathrm{e}^{-100t}\unit{}{\ampere}$。
    \end{answer}
\end{question}

\begin{question}
    图示电路中，试求 $\dot{U}$。(8 分)
    \begin{figure}[H]
    \hfill
    \input fig09
    \end{figure}
    \examvspace*{2cm}
    \begin{answer}
        $\dot{I} = 1\Varangle{\unit{0}{\degree}}(0.5 + \mathrm{j}0.5) = \sqrt{2} / 2
        \Varangle{\unit{45}{\degree}}$，$\dot{U} = \mathrm{j}1\dot{I} +
        1\Varangle{\unit{0}{\degree}} = \sqrt{2} / 2
        \Varangle{\unit{45}{\degree}}$。
    \end{answer}
\end{question}

\begin{question}
    已知某段电路的电压、电流分别为
    \begin{equation*}
        \begin{aligned}
            \dot{U} &= -10\Varangle{\unit{-150}{\degree}}\unit{}{\volt} \\
                  i &= 10\cos (200\pi t+\unit{340}{\degree})\unit{}{\ampere}
        \end{aligned}
    \end{equation*}
    (1)求它们的有效值、频率 $f$ 和周期 $T$。(2)写出它们的标准相量形式，
    求两者相位差，指出哪个相位超前。(3)画出两者的相量图。(7 分)
    \examvspace*{2cm}
    \begin{answer}
        $f=\unit{100}{\hertz}$，
        $\dot{U}=10\Varangle{\unit{30}{\degree}}\unit{}{\volt}$，
        $\dot{I}=5\sqrt{2}\Varangle{\unit{-20}{\degree}}\unit{}{\ampere}$
        。电压 $\dot{U}$ 相位超前 $\unit{50}{\degree}$。
    \end{answer}
\end{question}
